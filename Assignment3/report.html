We decided to work on a subset of the data, and use the range of years from 2020-2024, because we only have data for January 2025 on the ARNES cluster.

Based on these outputs we can observe a few differences, which are important for our further work and data ingestion part:

  • airport_fee column is named as airport_fee during years 2020-2023, but in 2024 it is named as Airport_fee.
  • another issue with airport_fee column is its type, which is object in 2020, but for 2021-2024 it is float64.
  • VendorID column is of type int64 during years 2020-2023 but in 2024 it is of type int32. Same issue happens with PULocationID and DOLocationID columns.

Now that we have selected the years we want to work with, and also inspected and identified the differences in file structure, we can run this data ingestion script, ingest.py. For running we used nohup python ingest.py > ingest_stdout.log 2>&1 &, so we can run it in background and also save the outputs in a log file.

def log(msg):
    print(msg, flush=True)
    with open(log_file, "a") as f:
        f.write(msg + "\n")


def step_done_file(name):
    return os.path.join(state_dir, f"{name}.done")


def is_step_done(name):
    return os.path.exists(step_done_file(name))


def mark_step_done(name):
    with open(step_done_file(name), "w") as f:
        f.write(f"done at {time.ctime()}\n")


def count_files_in_dir(path):
    if not os.path.exists(path):
        return 0
    total = 0
    for _, _, files in os.walk(path):
        total += len(files)
    return total


def count_matching_files(path, suffix):
    if not os.path.exists(path):
        return 0
    total = 0
    for f in os.listdir(path):
        if f.endswith(suffix):
            total += 1
    return total


def fix_for_hdf5(pdf):
    for col in pdf.columns:
        if str(pdf[col].dtype).startswith("string"):
            pdf[col] = pdf[col].astype("object")
    return pdf
  • Fixing the mentioned differences in schema structures between years. We also make sure that all expected columns are present in the data, and if there are some missing we add them with null values, so we have a consistent schema.
def fix_schema(df):
    if "Airport_fee" in df.columns:
        df = df.rename(columns={"Airport_fee": "airport_fee"})

    if "airport_fee" not in df.columns:
        df["airport_fee"] = 0.0

    if "VendorID" in df.columns:
        df["VendorID"] = df["VendorID"].astype("int64")

    if "PULocationID" in df.columns:
        df["PULocationID"] = df["PULocationID"].astype("int64")

    if "DOLocationID" in df.columns:
        df["DOLocationID"] = df["DOLocationID"].astype("int64")

    if "airport_fee" in df.columns:
        df["airport_fee"] = df["airport_fee"].astype("float64")

    if "passenger_count" in df.columns:
        df["passenger_count"] = df["passenger_count"].astype("float64")

    if "RatecodeID" in df.columns:
        df["RatecodeID"] = df["RatecodeID"].astype("float64")

    for col in expected_columns:
        if col not in df.columns:
            df[col] = None

    df = df[expected_columns]

    return df

def load_month(year, month):
    path = f"{base_path}/yellow_tripdata_{year}-{month:02d}.parquet"
    df = dd.read_parquet(path)
    df = fix_schema(df)
    df["year"] = year
    return df

def load_year(year):
    months = []

    for month in range(1, 13):
        path = f"{base_path}/yellow_tripdata_{year}-{month:02d}.parquet"
        if os.path.exists(path):
            log(f"Loading {year}-{month:02d}")
            df_month = load_month(year, month)
            months.append(df_month)

    df = dd.concat(months, interleave_partitions=True)
    return df

Comparing data sizes in different formats¶

To help us better understand differences between the formats and their sizes, we used this small script, check_output.py, where we checked the number of files and total size of them, and then also the average, minimum and maximum size of the files in each format.

import os

base = "/d/hpc/projects/FRI/bigdata/students/mz1034/A3/outputs"

paths = {
    "CSV 5y": os.path.join(base, "csv_2020_2024"),
    "CSV 2024 (partitioned)": os.path.join(base, "csv_2024"),
    "CSV 2024 (single file)": os.path.join(base, "csv_2024_single"),
    "HDF5 2024 (monthly)": os.path.join(base, "hdf5_2024"),
    "HDF5 2024 (single file)": os.path.join(base, "hdf5_2024_single"),
    "Parquet optimized": os.path.join(base, "parquet_optimized"),
}


def get_all_files(path):
    files = []
    for root, dirs, filenames in os.walk(path):
        for f in filenames:
            full = os.path.join(root, f)
            files.append(full)
    return files


def analyze(path):
    files = get_all_files(path)

    sizes = []
    for f in files:
        size = os.path.getsize(f) / (1024 * 1024)  # MB
        sizes.append(size)

    if len(sizes) == 0:
        return None

    return {
        "count": len(sizes),
        "total": sum(sizes),
        "min": min(sizes),
        "max": max(sizes),
        "avg": sum(sizes) / len(sizes),
    }


print("=" * 20)
print("SIZE ANALYSIS")
print("=" * 20)

for name, path in paths.items():
    print()
    print(name)
    print("-" * 20)

    result = analyze(path)

    if result is None:
        print("No files found")
        continue

    print(f"Files: {result['count']}")
    print(f"Total size (MB): {result['total']:.2f}")
    print(f"Min size (MB): {result['min']:.2f}")
    print(f"Max size (MB): {result['max']:.2f}")
    print(f"Avg size (MB): {result['avg']:.2f}")

We obtained the following results:

Parquet (optimized) 62 3399.71 0.01 115.17 54.83

From this we can see that the optimized parquet format is the smallest by far because of its columnar storage and the used compression. CSV is larger because it stores data in plain text and has no compression in use, meanwhile HDF5 is most likely larger than CSV because of the metadata and the structure overhead (the way it stores data in a hierarchical way). But in general we can see that parquet file is the most efficient format for working with big data.
